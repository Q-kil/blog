# UAParser
``` js
const uaInfo = new UAParser();
console.log("*******", uaInfo.getOS().name);
const device = uaInfo.getOS().name;
if (device == 'Android' || device == 'iOS') {
  if (virtual) return window.open( `g/${this.gameId}/m?virtual=${virtual}` );
  return window.open( `g/${this.gameId}/m` );
}
```
